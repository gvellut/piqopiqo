    def get_volume_info(volume_path: str) -> Optional[Dict[str, Any]]:
    """Get metadata for a volume using diskutil."""
    try:
        result = subprocess.run(
            ["diskutil", "info", "-plist", volume_path],
            capture_output=True,
            check=True
        )
        return plistlib.loads(result.stdout)
    except (subprocess.CalledProcessError, plistlib.InvalidFileException):
        return None

def get_media_type(volume_path: str) -> str:
    """
    Returns 'internal', 'sd_card', 'external', or 'unknown'
    """
    info = get_volume_info(volume_path)
    if not info:
        return "unknown"

    is_internal = info.get("Internal", False)
    protocol = info.get("Protocol", "")
    
    if is_internal:
        return "internal"
    
    # Common protocols for SD cards: 'SD', 'Secure Digital'
    if "SD" in protocol or "Secure Digital" in protocol:
        return "sd_card"
        
    return "external"
